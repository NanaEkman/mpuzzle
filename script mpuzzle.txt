create database mpuzzle;

use mpuzzle;

create table usuario (
ID integer auto_increment,
Nickname varchar(30),
Senha Varchar(30),
Email Varchar(30),
primary key(ID));

create table jogada(
ID integer auto_increment,
dificuldade integer,
idUsuario integer,
tempo Time,
rodada integer,
primary key(ID, dificuldade, tema, idUsuario, tempo, rodada),
foreign key(idUsuario) references usuario(ID));
